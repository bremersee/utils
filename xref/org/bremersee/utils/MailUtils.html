17  package org.bremersee.utils;
18  
19  import java.util.Collections;
20  import java.util.HashMap;
21  import java.util.HashSet;
22  import java.util.Map;
23  import java.util.Set;
24  
25  import org.apache.commons.lang3.Validate;

27  /**
28   * <p>
29   * Methods to work with email addresses (from {@code bremersee.org}).<br/>
30   * Be aware: Other mail domains / server may treat email addresses in a
31   * different way.
32   * </p>
33   * 
34   * @author Christian Bremer
35   */
36  public abstract class MailUtils {
37  
38      private MailUtils() {
39          // utility class, never constructed
40      }
41  
42      /**
43       * Simple email address regex that fits
44       * <a href="https://tools.ietf.org/html/rfc5322">RFC 5322</a> and
45       * <a href="https://tools.ietf.org/html/rfc6531">RFC 6531</a>.
46       */
47      public static final String EMAIL_REGEX = "^\\S+@\\S+$";
48  
49      /**
50       * Email address regex for {@code bremersee.org} (
51       * <a href="https://tools.ietf.org/html/rfc5322">RFC 5322</a>, but without
52       * {@code '%'}).<br />
53       * <a href="https://tools.ietf.org/html/rfc6531">RFC 6531</a> (characters
54       * above ASCII-Codes 127) is not supported.
55       */
56      public static final String BREMERSEE_EMAIL_REGEX = "[a-zA-Z0-9!#$&'*+/=?^_`{|}~-]+(?:\\.[a-zA-Z0-9!#$&'*+/=?^_`{|}~-]+)*@(?:[a-zA-Z0-9](?:[a-zA-Z0-9-]*[a-zA-Z0-9])?\\.)+[a-zA-Z0-9](?:[a-zA-Z0-9-]*[a-zA-Z0-9])?";
57  
58      /**
59       * A set of characters that are allowed in the local part of an email
60       * address from {@code bremersee.org}.
61       */
62      private static final Set<Character> LEGAL_CHARS_IN_MAIL_LOCAL_PART;
63  
64      /**
65       * A map of characters to replace some illegal characters in the local prt
66       * of an email address from {@code bremersee.org}.
67       */
68      private static final Map<Character, String> REPLACE_CHARS_IN_MAIL_LOCAL_PART_MAP;
69  
70      /**
71       * Build the map of illegal characters.
72       */
73      static {
74          Set<Character> tmpCharSet = new HashSet<Character>();
75          char n0 = '0';
76          char n9 = '9';
77          char A = 'A';
78          char Z = 'Z';
79          char a = 'a';
80          char z = 'z';
81          char[] x = { '#', '$', '&', '\'', '*', '+', '-', '/', '=', '?', '^', '_', '`', '{', '|', '}', '~', '.' };
82  
83          StringBuilder sb = new StringBuilder();
84          for (int i = (int) n0; i <= (int) n9; i++) {
85              sb.append((char) i);
86          }
87          for (int i = (int) A; i <= (int) Z; i++) {
88              sb.append((char) i);
89          }
90          for (int i = (int) a; i <= (int) z; i++) {
91              sb.append((char) i);
92          }
93          sb.append(x);
94          for (int i = 0; i < sb.length(); i++) {
95              tmpCharSet.add(sb.charAt(i));
96          }
97          LEGAL_CHARS_IN_MAIL_LOCAL_PART = Collections.unmodifiableSet(tmpCharSet);
98  
99          Map<Character, String> tmpReplaceCharsMap = new HashMap<Character, String>();
100         tmpReplaceCharsMap.put('(', "{");
101         tmpReplaceCharsMap.put(')', "}");
102         tmpReplaceCharsMap.put('[', "{");
103         tmpReplaceCharsMap.put(']', "}");
104         tmpReplaceCharsMap.put('\\', "/");
105 
106         tmpReplaceCharsMap.put('Ä', "Ae");
107         tmpReplaceCharsMap.put('Ö', "Oe");
108         tmpReplaceCharsMap.put('Ü', "Ue");
109 
110         tmpReplaceCharsMap.put('ä', "ae");
111         tmpReplaceCharsMap.put('ö', "oe");
112         tmpReplaceCharsMap.put('ü', "ue");
113 
114         tmpReplaceCharsMap.put('ß', "ss");
115 
116         tmpReplaceCharsMap.put('Á', "A");
117         tmpReplaceCharsMap.put('É', "E");
118         tmpReplaceCharsMap.put('Í', "I");
119         tmpReplaceCharsMap.put('Ó', "O");
120         tmpReplaceCharsMap.put('Ú', "U");
121 
122         tmpReplaceCharsMap.put('á', "a");
123         tmpReplaceCharsMap.put('é', "e");
124         tmpReplaceCharsMap.put('í', "i");
125         tmpReplaceCharsMap.put('ó', "o");
126         tmpReplaceCharsMap.put('ú', "u");
127 
128         tmpReplaceCharsMap.put('À', "A");
129         tmpReplaceCharsMap.put('È', "E");
130         tmpReplaceCharsMap.put('Ì', "I");
131         tmpReplaceCharsMap.put('Ò', "O");
132         tmpReplaceCharsMap.put('Ù', "U");
133 
134         tmpReplaceCharsMap.put('à', "a");
135         tmpReplaceCharsMap.put('è', "e");
136         tmpReplaceCharsMap.put('ì', "i");
137         tmpReplaceCharsMap.put('ò', "o");
138         tmpReplaceCharsMap.put('ù', "u");
139 
140         tmpReplaceCharsMap.put('Â', "A");
141         tmpReplaceCharsMap.put('Ê', "E");
142         tmpReplaceCharsMap.put('Î', "I");
143         tmpReplaceCharsMap.put('Ô', "O");
144         tmpReplaceCharsMap.put('Û', "U");
145 
146         tmpReplaceCharsMap.put('â', "a");
147         tmpReplaceCharsMap.put('ê', "e");
148         tmpReplaceCharsMap.put('î', "i");
149         tmpReplaceCharsMap.put('ô', "o");
150         tmpReplaceCharsMap.put('û', "u");
151 
152         REPLACE_CHARS_IN_MAIL_LOCAL_PART_MAP = Collections.unmodifiableMap(tmpReplaceCharsMap);
153     }
154 
155     /**
156      * Build an email address from the local part and the mail domain name.<br/>
157      * Illegal characters in the local part will be replaced by legal ones.<br/>
158      * The mail domain name will not be validated.<br/>
159      * So this method can be used to generate mail addresses from the name of a
160      * group, for example.
161      * 
162      * @param localPart
163      *            the local part of the email address
164      * @param mailDomain
165      *            the mail domain name
166      * @return a legal email address (the local part may has changed)
167      */
168     public static String buildMailAddress(String localPart, String mailDomain) {
169 
170     	Validate.notBlank(localPart, "localPart must not be null or blank");
171     	Validate.notBlank(mailDomain, "mailDomain must not be null or blank");
172 
173         StringBuilder localPartBuilder = new StringBuilder();
174         for (int i = 0; i < localPart.length(); i++) {
175             Character c = localPart.charAt(i);
176             if (LEGAL_CHARS_IN_MAIL_LOCAL_PART.contains(c)) {
177                 localPartBuilder.append(c);
178             } else {
179                 String r = REPLACE_CHARS_IN_MAIL_LOCAL_PART_MAP.get(c);
180                 if (r == null) {
181                     localPartBuilder.append('_');
182                 } else {
183                     localPartBuilder.append(r);
184                 }
185             }
186         }
187         return localPartBuilder.toString() + "@" + mailDomain;
188     }
