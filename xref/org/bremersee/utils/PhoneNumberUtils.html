17  package org.bremersee.utils;
18  
19  import org.apache.commons.lang3.Validate;
20  
21  /**
22   * *
23   * <p>
24   * Some methods to work with phone numbers.
25   * </p>
26   * 
27   * @author Christian Bremer
28   */
29  public abstract class PhoneNumberUtils {
30  
31      private PhoneNumberUtils() {
32          // utility class, never constructed
33      }
34  
35      /**
36       * Create a new phone number from the specified one that only contains
37       * numbers, '*' or '#'.
38       * 
39       * @param phoneNumber
40       *            the original phone number
41       * @return the cleaned phone number
42       */
43      public static String cleanPhoneNumber(String phoneNumber) {
44  
45          Validate.notEmpty(phoneNumber, "Phone number must not be null or blank.");
46  
47          phoneNumber = phoneNumber.trim();
48  
49          if (phoneNumber.startsWith("+")) {
50              phoneNumber = "00" + phoneNumber.substring(1);
51          }
52  
53          phoneNumber = phoneNumber.toUpperCase();
54          phoneNumber = phoneNumber.replace('A', '2');
55          phoneNumber = phoneNumber.replace('B', '2');
56          phoneNumber = phoneNumber.replace('C', '2');
57          phoneNumber = phoneNumber.replace('D', '3');
58          phoneNumber = phoneNumber.replace('E', '3');
59          phoneNumber = phoneNumber.replace('F', '3');
60          phoneNumber = phoneNumber.replace('G', '4');
61          phoneNumber = phoneNumber.replace('H', '4');
62          phoneNumber = phoneNumber.replace('I', '4');
63          phoneNumber = phoneNumber.replace('J', '5');
64          phoneNumber = phoneNumber.replace('K', '5');
65          phoneNumber = phoneNumber.replace('L', '5');
66          phoneNumber = phoneNumber.replace('M', '6');
67          phoneNumber = phoneNumber.replace('N', '6');
68          phoneNumber = phoneNumber.replace('O', '6');
69          phoneNumber = phoneNumber.replace('P', '7');
70          phoneNumber = phoneNumber.replace('Q', '7');
71          phoneNumber = phoneNumber.replace('R', '7');
72          phoneNumber = phoneNumber.replace('S', '7');
73          phoneNumber = phoneNumber.replace('T', '8');
74          phoneNumber = phoneNumber.replace('U', '8');
75          phoneNumber = phoneNumber.replace('V', '8');
76          phoneNumber = phoneNumber.replace('W', '9');
77          phoneNumber = phoneNumber.replace('X', '9');
78          phoneNumber = phoneNumber.replace('Y', '9');
79          phoneNumber = phoneNumber.replace('Z', '9');
80  
81          int x1 = (int) '*';
82          int x2 = (int) '#';
83  
84          int n0 = (int) '0';
85          int n9 = (int) '9';
86  
87          StringBuilder sb = new StringBuilder();
88          for (int i = 0; i < phoneNumber.length(); i++) {
89  
90              int n = (int) phoneNumber.charAt(i);
91              if (n == x1 || n == x2 || (n0 <= n && n <= n9)) {
92                  sb.append((char) n);
93              }
94          }
95  
96          return sb.toString();
97      }
98  
99  }
